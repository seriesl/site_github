Plotly example

Code 
import plotly.graph_objects as go
import numpy as np
Code 
x = np.linspace(0, 4*np.pi, 500)

fig = go.Figure()
fig.add_trace(go.Scatter(x=x, y=np.sin(x), name='sin(x)'))
fig.add_trace(go.Scatter(x=x, y=np.cos(x), name='cos(x)'))
fig.show()
